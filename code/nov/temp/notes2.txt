import { Component } from '@angular/core';
import { Router, RouterOutlet } from '@angular/router';
import { slideRouteAnimation } from './animations';

@Component({
  selector: 'app-parent',
  templateUrl: './parent.component.html',
  styleUrls: ['./parent.component.scss'],
  animations: [slideRouteAnimation]
})
export class ParentComponent {
  constructor(private router: Router) {}

  slideAndNavigate() {
    this.router.navigate(['/next-page']); // Use your real route
  }

  prepareRoute(outlet: RouterOutlet) {
    return outlet?.activatedRouteData?.['animation'];
  }
}

const routes: Routes = [
  {
    path: 'page-one',
    component: PageOneComponent,
    data: { animation: 'PageOne' }
  },
  {
    path: 'next-page',
    component: PageTwoComponent,
    data: { animation: 'PageTwo' }
  }
];



import {
  trigger,
  transition,
  style,
  query,
  animate,
  group
} from '@angular/animations';

export const slideRouteAnimation = trigger('routeAnimation', [
  transition('* <=> *', [
    query(':enter, :leave', [
      style({
        position: 'absolute',
        width: '100%',
        top: 0,
        left: 0
      })
    ], { optional: true }),

    group([
      // Leave page goes left
      query(':leave', [
        animate('400ms ease-in-out', style({ transform: 'translateX(-100%)' }))
      ], { optional: true }),

      // Enter page comes from right
      query(':enter', [
        style({ transform: 'translateX(100%)' }),
        animate('400ms ease-in-out', style({ transform: 'translateX(0%)' }))
      ], { optional: true })
    ])
  ])
]);


.page-area {

  position: relative;
  height: 100%;
}

.page-content {
